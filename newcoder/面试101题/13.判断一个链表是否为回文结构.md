# 13.判断一个链表是否为回文结构

[判断一个链表是否为回文结构](https://www.nowcoder.com/practice/3fed228444e740c8be66232ce8b87c2f?tpId=295&tqId=1008769&ru=/exam/oj&qru=/ta/format-top101/question-ranking&sourceUrl=%2Fexam%2Foj)



#### 递归：

```c++
class Solution {
public:
    ListNode* firstHead;
    bool isPalindrome(ListNode* head) {
        if(!head || !head->next) return true;
        firstHead = head;
        return check(head);
    }

    bool check(ListNode* head){
        if(!head) return true;
        if(!check(head->next)) return false;
        if(head->val != firstHead->val) return false;
        firstHead = firstHead->next;
        return true;
    }
};
```





#### 双指针：

```c++
class Solution {
public:
    bool isPail(ListNode* head) {
        ListNode* slow = head,*fast = head;
        while(fast && fast->next){
            slow = slow->next;
            fast = fast->next->next;
        }        	// slow此刻就是 链表的后一半 的起始节点
        if(fast){ 	// 奇数链表 slow 从中间移动到 后一半
            slow = slow->next;
        }
        slow = reverse(slow);
        while(slow){
            if(slow->val != head->val) return false;
            slow = slow->next;
            head = head->next;
        }
        return true;
    }
    
    // 反转链表  递归
    ListNode* reverse(ListNode* head){
        if(!head || !head->next) return head;
        ListNode* node = reverse(head->next);
        
        head->next->next = head;
        head->next = nullptr;
        return node;
    }
    
    // 反转链表  非递归
    ListNode* reverse1(ListNode* head){
        ListNode* pre = NULL;
        while(head != NULL){
            // 保存后继，并断开
            ListNode* next = head->next;
            head->next = pre;
            pre = head;
            head = next;
        }
        return pre;
    }
};
```



#### 栈：

```c++
class Solution {
public:
    bool isPail(ListNode* head) {
        stack<ListNode*> stk;
        ListNode* slow = head,*fast = head;
        while(fast && fast->next){
            slow = slow->next;
            fast = fast->next->next;
        }
        while(slow){
            stk.emplace(slow);
            slow = slow->next;
        }
        fast = head;
        while(!stk.empty()){
            if(fast->val != stk.top()->val) return false;
            stk.pop();
            fast = fast->next;
        }
        return true;
    }
};
```



#### 数组：&#x20;

```c++
class Solution {
public:
    bool isPail(ListNode* head) {
        vector<int> nums;
        while(head){
            nums.emplace_back(head->val);
            head = head->next;
        }
        
        int n = nums.size();
        
        int i = 0, j = n-1;
        while(i <= j){
            if(nums[i] == nums[j]){
                i++; j--;
            }else return false;
        }
        return true;
    }
};
```
