# 3.链表中的节点每k个一组翻转

[链表中的节点每k个一组翻转](https://www.nowcoder.com/practice/b49c3dc907814e9bbfa8437c251b028e?tpId=295&tags=&title=&difficulty=0&judgeStatus=0&rp=0&sourceUrl=%2Fexam%2Foj%3Fpage%3D1%26tab%3D%25E7%25AE%2597%25E6%25B3%2595%25E7%25AF%2587%26topicId%3D295)



#### 递归：

```c++
class Solution {
public:
    ListNode* reverseKGroup(ListNode* head, int k) {
        if(!head || k == 1) return head;
        ListNode* cur = head;
        // 找 k 个节点的最后一个节点
        for(int i = 1; i < k; ++i){
            cur = cur->next;
            if(!cur) return head;    // 不足k个直接返回 头结点
        }
        ListNode* nxt = cur->next;    // 下一个 head
        cur->next = nullptr;          // 断链
        nxt = reverseKGroup(nxt, k);  // 后续继续反转
        ListNode* newHead = reverse(head);    // 反转前k个得到新头结点
        head->next = nxt;                // 链接当前k个节点 和后续
        return newHead;
    }
    // 反转链表(原地反转整个链表)
    ListNode* reverse(ListNode* head){
        if(!head || !head->next) return head;
        ListNode* newHead = reverse(head->next);
        head->next->next = head;        // 后继指向前驱
        head->next = nullptr;        // 尾指针置空
        return newHead;
    }
};
```



#### 递归：

```c++
class Solution {
public:
    ListNode* reverseKGroup(ListNode* head, int k) {
        if(!head) return head;
        ListNode* cur = head;
        // 判断当前是否存在 k 个节点
        for(int i = 1; i < k;i++){
            if(cur) cur = cur->next;
        }
        ListNode* node = nullptr;
        // 如果cur存在，表示存在k个节点，直接递归即可。
        if(cur) node = reverseKGroup(cur->next, k);
        else return head;  // 不足k个节点直接返回 head
        
        // 链接上递归后的 链表
        cur->next = node;
        // 当前k个节点进行 反转即可。
        node = reverseEnd(head, k);
        return node;
    }
    
    // 从第一个节点翻转到 第n个节点
    ListNode* reverseEnd(ListNode* head, int n){
        if(n == 1 || !head || !head->next) return head;
        ListNode* res = reverseEnd(head->next, n - 1);
        
        ListNode* cur = head->next;
        head->next = cur->next;
        cur->next = head;
        return res;
    }
};
```



#### 迭代：

```c++
class Solution {
public:
    ListNode* reverseKGroup(ListNode* head, int k) {
        if(!head || k == 1) return head;
        ListNode* dummy = new ListNode(1);
        dummy->next = head;
        ListNode* cur = head;
        ListNode* tail = dummy;
        
        while(cur){
            ListNode* pHead = cur;        // 当前段 头结点
            // 找k个节点 进行反转
            for(int i = 1; i < k; ++i){
                if(cur) cur = cur->next; 
            }
            if(!cur) {
                tail->next = pHead;    // 链接不足k 的段
                break;   // 如果当前段 不足k个 直接跳出循环
            }
            ListNode* nxt = cur->next;    // 下一段开始
            cur->next = nullptr;        // 断链
            
            // 反转当前段,链接到之前反转好的
            tail->next = reverse(pHead);  
            tail = pHead;        // 新的尾结点 即，当前段 开始的头结点
            cur = nxt;
        }
        return dummy->next;
    }
    // 反转链表(原地反转整个链表)
    ListNode* reverse(ListNode* head){
        if(!head || !head->next) return head;
        ListNode* newHead = reverse(head->next);
        head->next->next = head;        // 后继指向前驱
        head->next = nullptr;        // 尾指针置空
        return newHead;
    }
};
```
