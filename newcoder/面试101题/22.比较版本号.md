#### 双指针：  遍历过程中 统计版本号 并比较

```c++
class Solution {
public:
    int compare(string version1, string version2) {
        int p1 = 0, p2 = 0;
        int len1 = version1.size(), len2 = version2.size();
        while(p1 < len1 || p2 < len2){
            int x = 0, y = 0;
            while(p1 < len1 && version1[p1] != '.'){
                x = x * 10 + (version1[p1++] - '0');   // 统计版本号数值，不存在则为0
            }
            ++p1; // 越过.
            while(p2 < len2 && version2[p2] != '.'){
                y = y * 10 + (version2[p2++] - '0');
            }
            ++p2; // 越过.
            if(x > y) return 1;
            if(x < y) return -1;
        }
        return 0;
    }    
};
```



#### 分割再比较：

```c++
class Solution {
public:
    int compare(string version1, string version2) {
        vector<int> v1,v2;
        preTask(version1,v1);
        preTask(version2,v2);
        int len1 = v1.size(), len2 = v2.size();
        int i = 0;
        while(i < len1 || i < len2){
            int x = 0, y = 0;
            x = i < len1 ? v1[i] : 0;
            y = i < len2 ? v2[i] : 0;
            if(x > y) return 1;
            else if(x < y) return -1;
            ++i;
        }
        return 0;
    }
    // 用 . 分割函数
    void preTask(string &s, vector<int>& res){
        string t = "";
        int n = s.size();
        for(int i = 0; i < n; ++i){
            if(isdigit(s[i])){
                t += s[i];
            }else if(s[i] == '.'){
                res.emplace_back(stoi(t));
                t.clear();
            }
        }
        res.emplace_back(stoi(t));
    }  
};
```



#### 使用 stringstream 用于分割

public:
    int compare(string version1, string version2) {
        vector<int> v1,v2;
        
    	stringstream ss;
    	ss << version1;

        ss.str(version2);

        int len1 = v1.size(), len2 = v2.size();

        while(i < len1 || i < len2){

        }
